% Bake underlines into the typeset text rather than relying on PDF annotation
% border rendering (which defaults to black — invisible on a dark background).
% \uline inherits the current color so the underline is blue, matching the text.
\makeatletter
\let\slughorn@href\href
\renewcommand{\href}[2]{\slughorn@href{#1}{\uline{#2}}}
\makeatother

% --- Code block styling: rounded corners, padding, inherited background --
% tcolorbox redefines Shaded at \AtBeginDocument so we win the load-order
% race against pandoc's own definition. colback=shadecolor inherits whatever
% --highlight-style sets, so swapping themes Just Works.
\usepackage{tcolorbox}
\tcbuselibrary{skins, breakable}

\AtBeginDocument{%
  \renewenvironment{Shaded}{%
    \begin{tcolorbox}[%
      enhanced,
      breakable,
      arc=4pt,
      outer arc=4pt,
      boxrule=0pt,
      colback=shadecolor,
      left=8pt, right=8pt, top=6pt, bottom=6pt,
      fontupper=\footnotesize
    ]%
  }{%
    \end{tcolorbox}%
  }%
}

% --- Blockquote callout: left bar only (Option A) ------------------------
% Wrapped in center so the 80%-wide box is horizontally centered.
% breakable is omitted — center creates a vbox that prevents page splits,
% and notes are almost always short enough that this doesn't matter.
\AtBeginDocument{%
  \renewenvironment{quote}{%
    \begin{center}%
    \begin{tcolorbox}[%
      enhanced,
      arc=0pt,
      outer arc=0pt,
      boxrule=0pt,
      leftrule=3pt,
      colback=pagebackground,
      colframe=linkblue,
      colupper=bodytext,
      width=0.8\linewidth,
      left=10pt, right=6pt, top=3pt, bottom=3pt,
      fontupper=\footnotesize,
    ]%
  }{%
    \end{tcolorbox}%
    \end{center}%
  }%
}

% --- H1 always starts on a fresh page -----------------------------------
